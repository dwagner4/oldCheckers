 22 /**
 23  * Callback for handling click events on the board
 24  *
 25  * @param {int} index index of testSpace
 26  * @param {string} code e | r | R | b | B
 27  */
 28 function handleBdClick ( index, code ) {
 29 
 30   // check for remote play and turn
 31   if ( match.redUID != match.blkUID && yourColor != match.turn.color) {
 32     console.log(" not your turn");
 33     return;
 34   }
 35 
 36   console.log("in the board click", index, code);
 37 
 38   if (code != "e") {  //not empty
 39     // is there a selected.piece piece already for this turn
 40     if (code == 'B') {code = 'b'}
 41     if (code == 'R') {code = 'r'}
 42     if (code == match.turn.color ) {
 43       if (selected == index) {
 44         selected = null;
 45       } else {
 46         selected = index;
 47       }
 48       allMoves = getPosSpaces(match)
 49       setBoard(match.boardState, selected, allMoves);
 50     } else {
 51       console.log("not selected.piece or wrong color ");
 52       return;
 53     }
 54   } else {
 55     if ( selected >= 0 ) {
 56       let rslt = doMove(match, selected, index);
 57       match = rslt.newMatchState;
 58       lastMove = rslt.lastMove;
 59       msg = rslt.msg;
 60       selected = null;
 61       allMoves = getPosSpaces(match)
 62       setBoard(match.boardState, selected, allMoves);
 63       console.log(msg);
 64     } else {
 65       console.log("empty space, no selected piece");
 66     }
 67   }
 68 }
