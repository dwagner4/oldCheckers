 19 /**
 20  * initiates a match object
 21  *
 22  * @param {string} turnColor  -'r' | 'b'
 23  * @param {string} redPlayerUID - 
 24  * @param {string} blkPlayerUID -
 25  * @return {object}  a match
 26  */
 27 function initMatch( turnColor,redPlayerUID,blkPlayerUID ) {
 28   let stat = "pending";
 29   if (redPlayerUID && blkPlayerUID) {  // local game?
 30     stat = "active";
 31   }
 32   return {turn: {color: turnColor, chain: false, chainSpace: null},
 33     boardState: startBoard,
 34     redUID: redPlayerUID,
 35     blkUID: blkPlayerUID,
 36     status: stat
 37   };
 38 }

149 /**
150  * calculate all possible next moves of player that has turn
151  *
152  * @param {object} match - object of current match,
153  * @return {array} array of int indexes
154  */
155  function getPosSpaces(match) {
156   let resultArray = []
157   let jmps = getAllJumps(match);
158   if (jmps.length > 0) {
159     for (var i = 0; i < jmps.length; i++) {
160       resultArray.push(jmps[i][0]);
161     }
162   } else {
163     let mvs = getAllMvs(match);
164     for (var i = 0; i < mvs.length; i++) {
165       resultArray.push(mvs[i][0]);
166     }
167   }
168   return resultArray;
169 }

175 // returns an array of all possible next moves
176 function getAllMvs(match) {
177   let moveSet = new Set();
178   let code = match.turn.color;
179   for (let i = 0; i < match.boardState.length; i++) {
180     let testPiece = match.boardState[i];
181     testCode = testPiece.toLowerCase();
182     if ( testCode == code ) {
183       let y = Math.floor( i / 4 );
184       let x = i % 4;
185       let odd = y % 2;
186       let testSpace = match.boardState[ i + 3 + odd ];
187       if (testPiece != 'b' && y < 7  && x + odd > 0 && testSpace == 'e') {
188         moveSet.add( [i + 3 + odd, i] );
189       }
190       testSpace = match.boardState[ i + 4 + odd ];
191       if (testPiece != 'b' && y < 7  &&  x + odd < 4 && testSpace == 'e') {
192         moveSet.add( [i + 4 + odd, i] );
193       }
194       testSpace = match.boardState[ i - 5 + odd ];
195       if (testPiece != 'r' && y > 0  && x + odd > 0 && testSpace == 'e') {
196         moveSet.add( [i - 5 + odd, i] );
197       }
198       testSpace = match.boardState[ i - 4 + odd ];
199       if (testPiece != 'r' && y > 0  && x + odd < 4 && testSpace == 'e') {
200         moveSet.add( [i - 4 + odd, i] );
201       }
202     }
203   }
204   let allMoves = [];
205   for (let move of moveSet) {
206     allMoves.push(move);
207   }
208   return allMoves;
209 }

211 //  returns an array of all possible jumps in the gameState
212 function getAllJumps(match) {
213   let jumpSet = new Set();
214   let code = match.turn.color;
215   for (let i = 0; i < match.boardState.length; i++) {
216     let testPiece = match.boardState[i];
217     testCode = testPiece.toLowerCase();
218     if ( testCode == code ) {
219 
220       let targetCode = 'b';
221       if (testCode == 'b') { targetCode = 'r'}
222       let y = Math.floor(i / 4);
223       let oddOffset = (y % 2)
224       let x = i % 4;
225 
226       let testSpace = match.boardState[ i + 7 ]
227       if ( testPiece != 'b' && y < 6  && testSpace == 'e' && x > 0 ){
228         captured = i + 3 + oddOffset;
229         if (match.boardState[captured].toLowerCase() == targetCode) {
230           jumpSet.add([i + 7, captured, i]);
231         }
232       }
233       testSpace = match.boardState[ i + 9 ]
234       if ( testPiece != 'b' && y < 6  && testSpace == 'e' && x < 3 ){
235         captured = i + 4 + oddOffset;
236         if (match.boardState[captured].toLowerCase() == targetCode) {
237           jumpSet.add([i + 9, captured, i]);
238         }
239       }
240       testSpace = match.boardState[ i - 9 ]
241       if( testPiece != 'r' && y > 1  && testSpace == 'e' && x > 0 ){
242         captured = i - 5 + oddOffset;
243         if (match.boardState[captured].toLowerCase() == targetCode) {
244           jumpSet.add([i - 9, captured, i]);
245         }
246       }
247       testSpace = match.boardState[ i - 7 ]
248       if ( testPiece != 'r' && y > 1  && testSpace == 'e' && x < 3 ){
249         captured = i - 4 + oddOffset;
250         if (match.boardState[captured].toLowerCase() == targetCode) {
251           jumpSet.add([i - 7, captured, i]);
252         }
253       }
254     }
255   }
256   let allJumps = [];
257   for (let jump of jumpSet) {
258     allJumps.push(jump);
259   }
260   return allJumps;
261 }
